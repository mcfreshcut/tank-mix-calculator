<body>
  <h1>Tank Mix Calculator</h1>
  <div class="card">
    <div class="row">
      <div>
        <label>Total tank size (gal)</label>
        <input id="tank" type="number" value="300" min="1" step="0.1">
      </div>
      <div>
        <label>Existing mix in tank (gal)</label>
        <input id="existing" type="number" value="50" min="0" step="0.1">
      </div>
    </div>
    <div class="row">
      <div>
        <label>Carrier rate (water) — gal/acre</label>
        <input id="carrier" type="number" value="1" min="0.1" step="0.1">
      </div>
      <div>
        <label>Target area (acres)</label>
        <input id="acres" type="number" value="1" min="0.01" step="0.01">
      </div>
    </div>

    <h3>Products</h3>
    <div class="row">
      <div>
        <label>Product A rate</label>
        <input id="aRate" type="number" value="2" step="0.1">
      </div>
      <div>
        <label>Units</label>
        <select id="aUnit">
          <option value="ozacre">oz/acre</option>
          <option value="ptacre">pt/acre</option>
          <option value="qtacre">qt/acre</option>
          <option value="galacre">gal/acre</option>
        </select>
      </div>
    </div>

    <p><b>Iron:</b> Fixed at 2 oz per 1,000 sq ft</p>

    <button id="calc" style="margin-top:12px;padding:10px 14px;">Calculate</button>
    <div id="out" class="card"></div>
    <p class="muted">Assumes 1 gal = 128 oz; 1 acre = 43,560 sq ft.</p>
  </div>

  <script>
    const U = { ptacre: 16, qtacre: 32, galacre: 128, ozacre: 1 };

    document.getElementById('calc').onclick = () => {
      const tank = +tankEl.value;
      const exist = +existing.value;
      const carrier = +carrierEl.value;
      const acres = +acresEl.value;
      const aRate = +aRateEl.value;
      const aUnit = aUnitEl.value;

      const galPerAcre = carrier;
      const totalWaterNeeded = galPerAcre * acres;
      const roomLeft = Math.max(0, tank - exist);

      // Product A
      const aRateOzPerAcre = aRate * (U[aUnit] || 1);
      const aTotalOz = aRateOzPerAcre * acres;
      const aTotalGal = aTotalOz / 128;

      // Iron (fixed at 2 oz/1k sq ft)
      const ironOzPerAcre = 2 * (43560 / 1000);
      const ironTotalOz = ironOzPerAcre * acres;
      const ironTotalGal = ironTotalOz / 128;

      const mixTotalGal = totalWaterNeeded + aTotalGal + ironTotalGal;

      const warn = [];
      if (mixTotalGal > tank) warn.push(`Mix exceeds tank size by ${(mixTotalGal - tank).toFixed(2)} gal.`);
      if (totalWaterNeeded > roomLeft) warn.push(`Not enough space for carrier: need ${totalWaterNeeded.toFixed(2)} gal water, room is ${roomLeft.toFixed(2)} gal.`);
      if (mixTotalGal > roomLeft + exist) warn.push(`Total planned mix (${mixTotalGal.toFixed(2)} gal) exceeds tank (${tank.toFixed(2)} gal).`);

      out.innerHTML = `
        <h3>Results</h3>
        <p><b>Add water:</b> ${totalWaterNeeded.toFixed(2)} gal</p>
        <p><b>Product A:</b> ${aTotalOz.toFixed(1)} oz (${aTotalGal.toFixed(2)} gal)</p>
        <p><b>Iron (fixed 2 oz/1k):</b> ${ironTotalOz.toFixed(1)} oz (${ironTotalGal.toFixed(2)} gal)</p>
        <p><b>Total mix volume:</b> ${mixTotalGal.toFixed(2)} gal</p>
        <p><b>Current tank:</b> ${exist.toFixed(2)} gal present → <b>Space left:</b> ${roomLeft.toFixed(2)} gal</p>
        ${warn.length ? `<div style="color:#b00;margin-top:8px;"><b>Warnings</b><ul>${warn.map(w=>`<li>${w}</li>`).join('')}</ul></div>` : ''}
      `;
    };

    const tankEl = document.getElementById('tank');
    const existing = document.getElementById('existing');
    const carrierEl = document.getElementById('carrier');
    const acresEl = document.getElementById('acres');
    const aRateEl = document.getElementById('aRate');
    const aUnitEl = document.getElementById('aUnit');
    const out = document.getElementById('out');
  </script>
</body>
